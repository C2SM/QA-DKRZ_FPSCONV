netcdf cf_6.1a {
dimensions:
   parcel = 2 ; // number of trajectories
   times = 3 ;
   max_len_parcel_name = 5 ; // max length of trajectory name

variables:
   float times(times) ;
      times:standard_name = "time" ;
      times:units="days since 1961-1-1";
   float lon(parcel,times) ;
      lon:standard_name = "longitude";
      lon:units= "degrees_east" ;
   float lat(parcel,times) ;
      lat:standard_name = "latitude";
      lat:units= "degrees_north" ;
   char parcel_name(max_len_parcel_name) ;

   float rh(parcel,times) ;
      rh:comment2 = "ordering of dimensions is valid, because times is not unlimited" ;
      rh:long_name = "relative humidity" ;
      rh:coordinates = "lat lon" ;
      rh:comment = "coordinates misses parcel_name" ;

// global attributes:
   :comment="Several parcel trajectories" ;
   :Conventions = "CF-1.4" ;

data:

times= 1., 2., 3. ;
parcel_name= "asdf" ;
}
