netcdf cf_7.3.3e {
dimensions:
   lat=2;
   lon=3;
   maxlen=20;
   ls=2;

variables:
   float asdf(ls,lat,lon);
      asdf:long_name = "surface temperature" ;
      asdf:coordinates="land_sea";
      asdf:cell_methods="area: mean where land_sea";
   float lat(lat) ;
      lat:units = "degrees_north" ;
      lat:standard_name = "latitude" ;
      lat:axis = "Y" ;
   float lon(lon) ;
      lon:units = "degrees_east" ;
      lon:standard_name = "longitude" ;
      lon:axis = "X" ;

   char land_sea(ls,maxlen);
      land_sea:standard_name="air_pressure";
      land_sea:units = "Pa" ;
      land_sea:comment = "invalid: standard_name must be area_type" ;

// global attributes:
   :comment="Mean surface temperature over land and sensible heat flux averaged separately over land and sea." ;
   :Conventions = "CF-1.4" ;

data:
 lat=1.,2.;
 lon=1.,2.,3.;
 land_sea="land","sea";
}
