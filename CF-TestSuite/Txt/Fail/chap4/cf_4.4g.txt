netcdf cf_4.4f {
dimensions:
   time = UNLIMITED ; // (4 currently)

variables:
   double time(time) ;
      time:long_name = "time" ;
      time:units = "days since 1990-1-1" ;

   double var(time) ;
      var:units="m" ;
      var:calendar="none";
      var:long_name="data variable";
      var:comment="calendar may only be attached to a time coordinate";

// global attributes:
   :Conventions = "CF-1.4" ;

data:
  time = 1., 2., 3., 4.;
  var= 1., 2., 3., 4.;
}
