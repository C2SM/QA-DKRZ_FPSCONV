netcdf cf_6.1d {
dimensions:
   parcel = 3 ; // number of trajectories
   times = 3 ;
   max_len_parcel_name = 5 ; // max length of trajectory name

variables:
   float times(times) ;
      times:standard_name = "time" ;
      times:units="days since 1961-1-1";
   float lon(parcel,times) ;
      lon:standard_name = "longitude";
      lon:units= "degrees_east" ;
   float lat(parcel,times) ;
      lat:standard_name = "latitude";
      lat:units= "degrees_north" ;
   float T(parcel,times) ;
      T:long_name = "temperature" ;
      T:coordinates = "parcel lat lon" ;
      T:comment = "order of dimensions is valid, because times is not unlimited" ;

   char parcel(parcel,max_len_parcel_name) ;
      parcel:comment="1) a label is an auxiliary, 2) not multi-dimensional coordinate" ;

// global attributes:
   :comment="Several parcel trajectories" ;
   :Conventions = "CF-1.4" ;

data:

times= 1., 2., 3. ;
parcel= "asdf", "qwer", "yxcv" ;
}
